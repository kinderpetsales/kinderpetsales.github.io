GEMINI_API_KEY = os.getenv("GEMINI_API_KEY")
GEMINI_MODEL = "gemini-2.5-flash"
GEMINI_URL = f"https://generativelanguage.googleapis.com/v1beta/models/{GEMINI_MODEL}:generateContent?key={GEMINI_API_KEY}"
question_history = {}

def call_gemini(prompt: str, user_id: str = None) -> str:
    """
    prompt: AI에게 보낼 메세지
    user_id: 대화 흐름을 기억하기 위한 사용자 ID (생략 시 단발성 질문으로 작동)
    """
    if not GEMINI_API_KEY: return "AI API 키가 설정되지 않았습니다."

    # 1. 대화 내역(History) 구성
    contents = []
    if user_id:
        if user_id not in question_history:
            question_history[user_id] = []
       
        # 과거 내역 불러오기
        contents.extend(question_history[user_id])
        # 새로운 질문 추가
        contents.append({"role": "user", "parts": [{"text": prompt}]})
    else:
        # user_id가 없으면 (예: 대기방 호출 등) 단발성 질문으로 처리
        contents = [{"role": "user", "parts": [{"text": prompt}]}]

    # API 요청 데이터 구성 (Google Search 도구 포함)
    payload = {
        "contents": contents,
        "tools": [{"googleSearch": {}}] # [핵심] 실시간 정보 검색 기능 활성화
    }

    try:
        # 검색 시간이 걸릴 수 있으므로 timeout을 30초로 늘렸습니다.
        resp = requests.post(GEMINI_URL, headers={"Content-Type": "application/json"}, data=json.dumps(payload), timeout=30)
        resp.raise_for_status()
       
        data = resp.json()
        ans_text = data["candidates"][0]["content"]["parts"][0]["text"].strip()

        # 2. 대화 내역 업데이트 (user_id가 있을 때만 저장)
        if user_id:
            # AI의 답변 기록 추가
            question_history[user_id].append({"role": "model", "parts": [{"text": ans_text}]})
            # 최신 왕복 5회(총 10개 메세지)만 유지하여 과도한 토큰 사용 방지
            if len(question_history[user_id]) > 10:
                question_history[user_id] = question_history[user_id][-10:]

        return ans_text
    except Exception as e:
        logger.error(f"Gemini API Error: {e}")
        return "⚠️ 답변 중 오류가 발생했습니다. 잠시 후 다시 시도해주세요."

def sale_api_url() -> str:
    return os.getenv("SALE_API_URL") or "https://ivqyxdtmyqopjlfaxdrj.supabase.co/rest/v1/sale"

def sale_api_key() -> str | None:
    return os.getenv("SALE_API_KEY")

def sale_site_url() -> str:

def sale_headers(include_json: bool = True) -> dict:
    apikey = sale_api_key()
    headers = {
        "apikey": apikey or "",
        "Authorization": f"Bearer {apikey}" if apikey else "",
        "Prefer": "return=representation",
    }
    if include_json:
        headers["Content-Type"] = "application/json"
    return headers

def sale_get(params: dict) -> tuple[bool, str, object]:
    apikey = sale_api_key()
    if not apikey: return False, "SALE_API_KEY 미설정(.env)", None
    url = sale_api_url()
    try:
        r = requests.get(url, headers=sale_headers(include_json=False), params=params, timeout=(3.0, 10.0))
        if 200 <= r.status_code < 300:
            try: return True, f"ok({r.status_code})", r.json()
            except Exception: return True, f"ok({r.status_code})", r.text
        return False, f"http {r.status_code}: {r.text[:300]}", None
    except Exception as e: return False, f"exception: {e}", None

def extract_kor_nickname_from_profile_line(profile_line: str) -> str | None:
    m = re.match(r"^([가-힣]{1,5})", (profile_line or "").strip())
    return m.group(1) if m else None

def sale_find_latest_row_by_profile_line(profile_line: str) -> tuple[bool, str, dict | None]:
    ok, err, data = sale_get({
        "select": "id,nick_name,valid_month,type,specification,link",
        "nick_name": f"eq.{(profile_line or '').strip()}",
        "order": "id.desc",
        "limit": "1",
    })
    if ok and isinstance(data, list) and data and isinstance(data[0], dict):
        return True, "ok", data[0]
    return False, err or "not found", None

def sale_find_latest_row_by_nickname_prefix(nickname: str) -> tuple[bool, str, dict | None]:
    nn = (nickname or "").strip()
    if not nn: return False, "empty nickname", None
    
    # 1. API 호출 시 limit을 늘려, 비슷한 이름을 가진 여러 후보를 가져옵니다.
    ok, err, data = sale_get({
        "select": "id,nick_name,valid_month,type,specification,link",
        "nick_name": f"ilike.{nn}*",
        "order": "id.desc",
        "limit": "10", 
    })
    
    # 2. 파이썬 단에서 정규식으로 정확한 경계(Boundary) 검사를 수행합니다.
    if ok and isinstance(data, list) and data:
        import re
        # 핵심: 추출한 이름(nn) 바로 뒤에 다른 '한글'이 오지 않는 경우만 본인으로 인정
        exact_match_pattern = re.compile(rf"^{nn}(?![가-힣])") 
        
        for row in data:
            db_nickname = row.get("nick_name", "")
            if exact_match_pattern.match(db_nickname):
                return True, "ok", row
                
    return False, err or "not found", None

def sale_compute_valid_month(now=None) -> int:
    try:
        from zoneinfo import ZoneInfo
        now = now or datetime.now(ZoneInfo("Asia/Seoul"))
    except Exception:
        now = now or datetime.now()
    if now.day <= 25: return int(now.month)
    return 1 if int(now.month) == 12 else int(now.month) + 1

def infer_sale_type_from_profile_line(profile_line: str) -> str | None:
    parts = (profile_line or "").split("/")
    role = parts[1].strip().upper() if len(parts) >= 2 else ""
    if role.startswith("D"): return "cat"
    if role.startswith("S"): return "dog"
    return None

def build_sale_payload_from_form(form_msg: str, chosen_text: str | None) -> tuple[dict, str | None]:
    try:
        from zoneinfo import ZoneInfo
        now = datetime.now(ZoneInfo("Asia/Seoul"))
    except Exception:
        now = datetime.now()

    m = re.search(r"단톡방 프로필 풀네임\s*->\s*([^\n]+)", form_msg, re.I)
    profile_line = m.group(1).strip() if m else ""

    m = re.search(r"키/몸무게\s*->\s*([^\n]+)", form_msg, re.I)
    hw_line = m.group(1).strip() if m else ""
    specification = hw_line if hw_line else None

    memo = ""
    m = re.search(r"B\.\s*분양원하는조건.*?->\s*(.*?)(?:\n\s*C\.|\n\s*C\.)", form_msg, re.I | re.S)
    if m: memo = m.group(1).strip()

    m = re.search(r"(https?://open\.kakao\.com/[^\s]+)", form_msg, re.I)
    link = m.group(1).strip() if m else ""

    inferred_type = None
    if chosen_text == "고양이": inferred_type = "cat"
    elif chosen_text == "강아지": inferred_type = "dog"
    else: inferred_type = infer_sale_type_from_profile_line(profile_line)

    payload = {
        "nick_name": profile_line,
        "specification": specification,
        "memo": memo,
        "link": link,
        "type": inferred_type,
        "valid_month": sale_compute_valid_month(now),
    }
    return payload, inferred_type

def sale_monthly_Maintenance_tick(bot_app) -> None:
    from Main import OPER_ROOM_ID
    from datetime import datetime
    try:
        from zoneinfo import ZoneInfo
        now = datetime.now(ZoneInfo("Asia/Seoul"))
    except Exception:
        now = datetime.now()

    if now.day != 1: return

    today_key = now.strftime("%Y-%m-%d")
    last_key = getattr(bot_app, "sale_last_cleanup_date", None)
    if last_key == today_key: return

    prev_month = 12 if now.month == 1 else (now.month - 1)
    ok, err, data = sale_delete_by_valid_month(prev_month)

    deleted_count = 0
    if ok and isinstance(data, list):
        deleted_count = len(data)

    try:
        if ok:
            bot_app.bot.api.reply(
                OPER_ROOM_ID,
                f"🧹[분양센터 자동정리]\n전월({prev_month}월) 분양글 삭제 완료\n삭제건수: {deleted_count}"
            )
        else:
            bot_app.bot.api.reply(
                OPER_ROOM_ID,
                f"🧹[분양센터 자동정리 실패]\n전월({prev_month}월) 삭제 실패\n사유: {err}"
            )
    except Exception: pass

    bot_app.sale_last_cleanup_date = today_key

def start_sale_Maintenance_scheduler(bot_app) -> None:
    import threading
    from datetime import datetime, timedelta
    try:
        from zoneinfo import ZoneInfo
        tz = ZoneInfo("Asia/Seoul")
    except Exception:
        tz = None

    if getattr(bot_app, "_sale_Maint_thread_started", False): return
    bot_app._sale_Maint_thread_started = True

    def _loop():
        try: sale_monthly_Maintenance_tick(bot_app)
        except Exception: pass

        while True:
            try:
                now = datetime.now(tz) if tz else datetime.now()
                next_run = (now + timedelta(days=1)).replace(hour=0, minute=1, second=0, microsecond=0)
                sleep_sec = (next_run - now).total_seconds()
                if sleep_sec < 1: sleep_sec = 1
            except Exception:
                sleep_sec = 60

            threading.Event().wait(sleep_sec)

            try: sale_monthly_Maintenance_tick(bot_app)
            except Exception: pass

    t = threading.Thread(target=_loop, name="sale_Maintenance_scheduler", daemon=True)
    t.start()

def handle_gyomu_gemini_chat(bot_app, ctx, msg: str) -> None:
    if bot_app.gyomu_global_gemini_disabled:
        return

    user_id = ctx.sender.id
    mode = bot_app.gyomu_mode.get(user_id, "menu")
    author_profile = ctx.sender.name or ""
    text = msg.strip()
    if not text:
        return

    if mode == "out_reason":
        handle_gyomu_out_reason_chat(bot_app, ctx, text, author_profile)
        return

    history = bot_app.gyomu_chat_history.get(user_id, [])
    history.append({"role": "user", "text": text})
    if len(history) > 10:
        history = history[-10:]

    log_lines = []
    for turn in history:
        role_kor = "사용자" if turn["role"] == "user" else "AI"
        log_lines.append(f"{role_kor}: {turn['text']}")

    base_prompt = f"""
너는 카카오톡 오픈채팅방의 AI 상담사야.
상담사는 친절하고 상냥하게 대하되, 간략하게 응대해.

현재 대화 중인 사용자의 프로필(닉네임): "{author_profile}"

- 사용자를 직접 부를 때는 반드시 위 닉네임을 대괄호 안에 그대로 포함해서 불러.
예: "[{author_profile}] 님, ~~~", "[{author_profile}] 님 말씀처럼 ~~~"
- 상담의 최종 목적은 사용자가 아래 1~6번 메뉴 중 하나를 스스로 선택해서
  채팅창에 [숫자만] 입력하도록 자연스럽게 유도하는 거야.
상담 메뉴:
1: 분양센터 관련 (분양 신청/등록,수정·삭제, 분양 관련 문의 등)
2: 외출/복귀 관련 (잠깐 나가있기, 외출 후 복귀 등)
3: 지인 초대 관련 (친구/지인 초대, 초대 링크 공유 등)
4: 기타 문의 및 제보/신고 (불편사항 제보, 궁금한 점, 운영진에게 질문 등)
5: 퇴학 후 문의/재입학 (왜 강퇴·퇴학당했는지, 다시 들어오고 싶은 경우 등)
6: 초대받고 온 경우 (지인에게 초대받아서 들어온 사람 등)

규칙:
- 인사(예: "안녕하세요")가 들어오면 먼저 반갑게 인사하고 어떤 용무로 방문했는지 자연스럽게 물어봐.
- 사용자의 말을 충분히 들어주고, 질문을 섞어서 자유롭게 대화해.
- 사용자의 상황이 어떤 메뉴(1~6)에 가장 가까운지 조용히 판단해서, "이 경우에는 5번 퇴학 후 문의가 어울릴 것 같아요. 채팅창에 5 라고만 입력해 주실래요?"
처럼 부드럽게 설명하고 번호 입력을 제안해.
- 한 번의 답변은 6줄 이하의 자연스러운 한국어 문장으로만 작성해.
- 숫자만 따로 출력하지 말고, 항상 문장 안에서 자연스럽게 숫자를 섞어서 말해.
"""

    prompt = base_prompt + "\n\n[지금까지의 대화 로그]\n" + "\n".join(log_lines) + "\n\n위 대화를 이어서, 이제 AI 상담사로서 다음 한 번의 답변만 작성해."

    try:
        ai_reply = call_gemini(prompt)
    except Exception as e:
        logger.error("교무실 Gemini 상담 호출 오류: %s", e, exc_info=True)
        fallback = gyomu_simple_intent_router(text)
        ctx.reply(fallback)
        return

    if not ai_reply or not ai_reply.strip():
        fallback = gyomu_simple_intent_router(text)
        ctx.reply(fallback)
        return

    ai_reply = ai_reply.strip()
    ctx.reply(ai_reply)

    history.append({"role": "assistant", "text": ai_reply})
    if len(history) > 10:
        history = history[-10:]
    bot_app.gyomu_chat_history[user_id] = history

def handle_gyomu_message(bot_app: Any, ctx: ChatContext, text: str) -> bool:

    if ctx.room.id != GYOMU_ROOM_ID:
        return False
   
    msg = text
    text = msg.strip()
    author_profile = ctx.sender.name or ""
    user_id = ctx.sender.id

    handle_gyomusil_admin_notify(bot_app, ctx, text)
    try: sale_monthly_Maintenance_tick(bot_app)
    except Exception: pass

    blocked_words = [
        '섹시','가슴','야하다','야해','야한','흥분','돔','섭','바텀','엉덩이','오랄','섹스',
        '형님','아다','보지','고추','꼬추','자지','디그','애널','정조대','발정','금딸','주인님',

        '피스팅','요도','복종','ㄷㄱㄹㅇㄷ','온플','영딸','조련','후빨','전립선','야하게',
        '야하더','야했어','야했더','야했습','형아','야노','야동','변태'
    ]
    for w in blocked_words:
        if w in msg:
            ctx.reply('⛔사용할 수 없는\n단어가 포함되었습니다.\n수정 후 다시보내주세요.')
            return True

    current_mode = bot_app.gyomu_mode.get(user_id)

    if text == "취소" and current_mode and current_mode.startswith("adopt_qna_"):
        bot_app.gyomu_mode.pop(user_id, None)
        if hasattr(bot_app, "gyomu_temp_adopt_data"):
            bot_app.gyomu_temp_adopt_data.pop(user_id, None)
        ctx.reply("진행 중인 분양센터 등록을 취소했습니다. 처음 메뉴로 돌아갑니다.")
        return True
    
    if current_mode == "return_verify":
        input_profile = text.strip()
        
        # 취소 처리
        if input_profile == "취소":
            ctx.reply("복귀 확인을 취소합니다.")
            bot_app.gyomu_mode.pop(user_id, None)
            return True

        # 2차 검증: 이름으로 찾기
        matched_uid, stored_name = bot_app.out_db.find_user_by_profile(input_profile)

        if matched_uid:
            # ✅ 2차 검증 성공 시에만 알림 전송
            _send_return_success_notifications(bot_app, ctx, stored_name, user_id)
        else:
            # ❌ 최종 실패 시에만 운영진에게 도움 요청 알림
            ctx.reply(f"죄송합니다. [{input_profile}] 프로필로 신청하신 외출 정보를 찾지 못했습니다.\n\n잠시 기다려주시면 운영진이 확인 후 응대해 드릴게요.")
            
            from Main import OPER_ROOM_ID
            admin_msg = f"📢 복귀 확인 최종 실패\n입력한 이름: {input_profile}\n현재 닉네임: {author_profile}\n(명단에 기록이 없어 확인이 필요합니다)"
            try:
                bot_app.bot.api.reply(OPER_ROOM_ID, admin_msg)
            except: pass

        bot_app.gyomu_mode.pop(user_id, None)
        return True
    
    elif current_mode == "adopt_qna_step1_confirm":
        if text == "예":
            # 💡 [디테일 1] 사용자가 "예"라고 했어도, 현재 프로필 양식이 규정에 맞는지 한 번 더 검사해야 합니다.
            profile_line = author_profile
            nick_match = re.match(r"^([가-힣]{1,5})", profile_line)
            nickname = nick_match.group(1) if nick_match else None
            
            # 닉네임이 없거나 형식이 이상한 경우 수동 입력으로 넘김
            if not nickname or len(profile_line) <= len(nickname):
                bot_app.gyomu_mode[user_id] = "adopt_qna_step1"
                ctx.reply(f"현재 프로필({profile_line}) 양식이 올바르지 않아 자동 입력이 불가합니다.\n\n등록하시려는 [단톡방 풀네임]을 직접 말씀해주세요.\n예시) 댕냥이/D/95/서울")
                return True
            
            # 숫자와 영문자 포함 여부 검사
            after = profile_line[len(nickname):len(nickname) + 15]
            has_num = any(c.isdigit() for c in after)
            has_eng = any("a" <= c.lower() <= "z" for c in after)
            if not (has_num and has_eng):
                bot_app.gyomu_mode[user_id] = "adopt_qna_step1"
                ctx.reply(f"현재 프로필({profile_line}) 양식이 올바르지 않아 자동 입력이 불가합니다.\n\n등록하시려는 [단톡방 풀네임]을 직접 말씀해주세요.\n예시) 댕냥이/D/95/서울")
